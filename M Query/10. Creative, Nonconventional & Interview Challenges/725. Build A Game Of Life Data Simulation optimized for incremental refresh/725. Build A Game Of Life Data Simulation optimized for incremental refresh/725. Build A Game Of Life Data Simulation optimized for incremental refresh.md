    // Define Game of Life rules
    isAlive = (cell) => if cell = 1 then true else false,
    countNeighbors = (grid, row, col) =>
        let
            neighbors = {
                {row-1, col-1}, {row-1, col}, {row-1, col+1},
                {row, col-1},             {row, col+1},
                {row+1, col-1}, {row+1, col}, {row+1, col+1}
            };
            List.Sum(List.Transform(neighbors, each if _ mod List.Count(grid) >= 0 and _ mod List.Count(grid[0]) >= 0 and grid[_[0]][_[1]] = 1 then 1 else 0))
        end,
    nextGeneration = (grid) =>
        let
            rows = List.Count(grid),
            cols = List.Count(grid[0]);
            List.Transform({0..rows-1}, each List.Transform({0..cols-1}, each
                let
                    alive = grid[_][__],
                    neighbors = countNeighbors(grid, _, __);
                in
                    if alive and (neighbors = 2 or neighbors = 3) then 1
                    else if not alive and neighbors = 3 then 1
                    else 0))
        end;
    // Run simulation for 10 generations
    generations = List.ToTable(List.Transform({0..10}, each nextGeneration(grid)))
    in
        generations
